How to Use SVG Icons with nz-icon

There are three main methods to use SVG icons with nz-icon:

1. Register SVG icon directly as a string

// In your component
import {{ '{' }} Component, OnInit {{ '}' }} from '@angular/core';
import {{ '{' }} NzIconService {{ '}' }} from 'ng-zorro-antd/icon';

@Component({{ '{' }}
  // component metadata
{{ '}' }})
export class YourComponent implements OnInit {{ '{' }}
  constructor(private iconService: NzIconService) {{ '{' }}{{ '}' }}

  ngOnInit(): void {{ '{' }}
    this.iconService.addIconLiteral('custom:star', '<svg viewBox="0 0 24 24">...</svg>');
  {{ '}' }}
{{ '}' }}

// In your template
<span nz-icon nzType="custom:star"></span>

2. Register SVG icon from iconfont

// In your component
import {{ '{' }} Component, OnInit {{ '}' }} from '@angular/core';
import {{ '{' }} NzIconService {{ '}' }} from 'ng-zorro-antd/icon';

@Component({{ '{' }}
  // component metadata
{{ '}' }})
export class YourComponent implements OnInit {{ '{' }}
  constructor(private iconService: NzIconService) {{ '{' }}{{ '}' }}

  ngOnInit(): void {{ '{' }}
    this.iconService.fetchFromIconfont({{ '{' }}
      scriptUrl: 'https://at.alicdn.com/t/font_8d5l8fzk5b87iudi.js'
    {{ '}' }});
  {{ '}' }}
{{ '}' }}

// In your template
<span nz-icon nzIconfont="icon-tuichu"></span>

3. Register SVG icon from assets folder

// In your component
import {{ '{' }} Component, OnInit {{ '}' }} from '@angular/core';
import {{ '{' }} NzIconService {{ '}' }} from 'ng-zorro-antd/icon';
import {{ '{' }} HttpClient {{ '}' }} from '@angular/common/http';

@Component({{ '{' }}
  // component metadata
{{ '}' }})
export class YourComponent implements OnInit {{ '{' }}
  constructor(
    private iconService: NzIconService,
    private http: HttpClient
  ) {{ '{' }}{{ '}' }}

  ngOnInit(): void {{ '{' }}
    this.http.get('assets/icons/star.svg', {{ '{' }} responseType: 'text' {{ '}' }})
      .subscribe((svg) => {{ '{' }}
        this.iconService.addIconLiteral('asset:star', svg);
      {{ '}' }});
  {{ '}' }}
{{ '}' }}

// In your template
<span nz-icon nzType="asset:star"></span>
